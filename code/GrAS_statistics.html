In [3]:
#Open rasters
neon_ndvi_rast = gdal.Open(neon_ndvi_fn)
ls8_ndvi_rast = gdal.Open(ls8_ndvi_fn)
neon_lai_rast = gdal.Open(neon_lai_fn)
#Get the dataset dimensions
cols = neon_ndvi_rast.RasterXSize
rows = neon_ndvi_rast.RasterYSize
#Read into an array
neon_ndvi_array = neon_ndvi_rast.GetRasterBand(1).ReadAsArray(0,0,cols,rows).astype(np.float)
ls8_ndvi_array = ls8_ndvi_rast.GetRasterBand(1).ReadAsArray(0,0,cols,rows).astype(np.float)
neon_lai_array = neon_lai_rast.GetRasterBand(1).ReadAsArray(0,0,cols,rows).astype(np.float)

Remove nan values¶

Do this by creating a new array that will store NaN values of both the NEON and LS 8 data

In [4]:
neon_ndvi_array_noNan = neon_ndvi_array.copy()
ls8_ndvi_array_noNan = ls8_ndvi_array.copy()
neon_lai_array_noNan = neon_lai_array.copy()
na_index = np.array(np.zeros((rows,cols)), dtype=float)
na_index[np.isnan(neon_ndvi_array_noNan)] = 1
na_index[np.isnan(ls8_ndvi_array_noNan)] = 1
na_index[np.isnan(neon_lai_array_noNan)] = 1
neon_ndvi_array_noNan = neon_ndvi_array_noNan[na_index==0]
ls8_ndvi_array_noNan = ls8_ndvi_array_noNan[na_index==0]
neon_lai_array_noNan = neon_lai_array_noNan[na_index==0]

Calculate Root Mean Square Error¶

In [6]:
rms = sqrt(mean_squared_error(neon_ndvi_array_noNan, ls8_ndvi_array_noNan))
print('Root mean square error = ' + np.str(np.round(rms,decimals=2)))

Root mean square error = 0.1

Calculate Pearson correlation coefficient¶

In [8]:
corr_coeff = pearsonr(neon_ndvi_array_noNan, ls8_ndvi_array_noNan)
print('Correlation coefficient = ', np.round(corr_coeff[0],decimals=2))   
print('p-value = ', np.round(corr_coeff[1],decimals=2))

Correlation coefficient =  0.26
